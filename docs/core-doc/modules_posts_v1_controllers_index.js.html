/**
 * @module utils/math
 * @description A handful of math helpers.
 */
function PostController(service) {
    return {
        list: async (req, reply) => {
            const results = await service.list(req.query); //
            return reply.respond().fetched(results, 'Posts retrieved');;
        },

        get: async (req, reply) => {
            const result = await service.get(req.params.id);
            return reply.respond().fetched(result, 'Post retrieved');
        },

        create: async (req, reply) => {
            const result = await service.create(req.body);
            return reply.respond().created(result,'Post created')
        },

        update: async (req, reply) => {
            const result = await service.update(req.params.id, req.body);
            return reply.respond().updated(result,"Post updated");
        },

        delete: async (req, reply) => {
            await service.delete(req.params.id);
            return reply.respond().deleted('Post deleted');
        },

        async searchByName(request, reply) {
            const result = await service.searchByName(request.query.name);
            return reply.respond().success(result,"fetching success");
        }
    };
}

module.exports = PostController;
